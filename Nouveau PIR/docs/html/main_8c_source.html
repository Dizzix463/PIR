665
666void __attribute__((__interrupt__, no_auto_psv)) TIM7_IRQHandler(void)
667{
668
669 if (failure_identify == '5') // CAM_PER --> Cam_Spk
670 {
671 if (GPIO_ReadInputDataBit(GPIOG, 8) == 1)
672 {
673 GPIO_ResetBits(GPIOG, 8);
674 }
675 else
676 {
677 GPIO_SetBits(GPIOG, 8);
678 };
679
680 counter_CAM_PER[1]++; // Number of times we lost CAM with timer 7
681 if (counter_CAM_PER[1] == 2)
682 {
683
684 TIM_Cmd(TIM7, DISABLE);
685
686 counter_CAM_PER[1] = 0; // Reset timer 7 CAM counter
687 }
688 TIM_SetCounter(TIM7, 0);
689 }
690 else if (failure_identify == '6') // CRK_TOOTH_PER
691 {
692 if (GPIO_ReadInputDataBit(GPIOG, 6) == 1)
693 {
694 GPIO_ResetBits(GPIOG, 6);
695 }
696 else
697 {
698 GPIO_SetBits(GPIOG, 6);
699 };
700
701 TIM_Cmd(TIM7, DISABLE);
702 }
703 else if (failure_identify == 'j')
704 { // SEG_ADP_ER_LIM
705
706 timer_Counter_SEG_ADP_ER_LIM++;
707 switch (timer_Counter_SEG_ADP_ER_LIM)
708 {
709 case 1: { // failure for the falling edge
710 GPIO_ResetBits(GPIOG, 6);
711 if (failure_waiting == true)
712 { // if the rising edge has already happen
713 if (sensortype_CRK == 'c') // sensor is cpdd
714 {
715 TIM_PrescalerConfig(TIM8, 0, TIM_PSCReloadMode_Immediate);
716 TIM_SetAutoreload(TIM8, 3599); //? 72*50=3600
717 }
718 else if (sensortype_CRK == 'h')
719 { // all the others
720 TIM_PrescalerConfig(TIM8, 0, TIM_PSCReloadMode_Immediate);
721 TIM_SetAutoreload(TIM8, (T_TOOTH_RAW / 2) - 1); //? If setted period is more than 910 us PSC should be set higher
722 }
723 else
724 {
725 // if sensor type not set, shouldn't happen
726 TIM_PrescalerConfig(TIM8, 0, TIM_PSCReloadMode_Immediate);
727 TIM_SetAutoreload(TIM8, 1);
728 }
729
730 // Start timer 8
731 TIM_Cmd(TIM8, ENABLE);
732 }
733 failure_passed = true;
734 break;

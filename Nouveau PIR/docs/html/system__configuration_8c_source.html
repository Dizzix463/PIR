58 // ## CRK Configuration
59 void CRK_configuration(void)
60 {
61 // Number of teeth between two gaps
62 number_teeth_between_gaps = (number_teeth_CRK/number_gap) - number_miss_teeth;
63
64 // Number of remaining teeth
65 number_real_teeth = number_teeth_CRK - number_miss_teeth;
66
67 // Revolution of one tooth time
68 revolution_CRK = (double)(360.0/number_teeth_CRK);
69
70 // Set number of segments
71 number_segments_CRK = number_cylinder;
72
73 int first_tooth_angle = 720 - tdc_to_gap;
74
75 if (first_seg_angle >= first_tooth_angle){
76 first_seg_start_tooth = (first_seg_angle - first_tooth_angle) / revolution_CRK;
77
78 }else{
79 first_seg_start_tooth = number_teeth_CRK - (( first_tooth_angle - first_seg_angle) / revolution_CRK);
80 }
81
82 // Set configuration status
83 CRK_config = true;
84
85 // Complete configuration
86 if((CRK_config == true) && (CAM_config == true))
87 {
88 CAM_CRK_configuration();
89 }
90
91 // Send configuration status
92 UART_send(message[1]);
93 }

96 // ## CAM Configuration
97 void CAM_configuration(void)
98 {
99 int camId = 0;
100
101 for(camId = 0; camId < number_of_CAM; camId++)
102 {
103 // Set counter to calculate the distances of the CAM edges to the gap in the for-loop
104 int i;
105
106 // Set number of CAM edges
107 if(active_CAM_edges[camId] == 'b')
108 {
109 number_teeth_CAM[camId] = number_active_edges_CAM[camId] / 2;
110 }
111 else
112 {
113 number_teeth_CAM[camId] = number_active_edges_CAM[camId];
114 }
115
116 // Calculate distance from gap to CAM edges
117 for(i = 0; i < number_active_edges_CAM[camId]; i++)
118 {
119 distance_gap_to_CAM_edge[camId][i] = CAM_edges[camId][i] + tdc_to_gap;
120
121 if(distance_gap_to_CAM_edge[camId][i] >= 720.0)
122 {
123 distance_gap_to_CAM_edge[camId][i] = distance_gap_to_CAM_edge[camId][i] - 720.0;
124 }
125 }
126 }
127
128
129 // Set configuration status
130 CAM_config = true;
131
132 // Complete configuration
133 if(CRK_config == true && CAM_config == true)
134 {
135 CAM_CRK_configuration();
136 }
137
138 // Send configuration status
139 UART_send(message[2]);
140
141 }

144 // ## CAM_CRK Configuration
145 void CAM_CRK_configuration(void)
146 {
147 int camId = 0;
148
149 for(camId = 0; camId < number_of_CAM; camId++)
150 {
151 // Set first falling(rising, when active_edges = 'r') CAM edge after TDC0
152 first_edge_to_tdc0[camId] = CAM_edges[camId][0];
153
154 // Distance from gap to first falling CAM-edge
155 gap_to_first_edge[camId] = first_edge_to_tdc0[camId] + tdc_to_gap;
156
157 if((long)tdc_to_gap%(long)revolution_CRK == 0)
158 {
159 TDC_on_edge[camId] = true;
160 }
161 else
162 {
163 TDC_on_edge[camId] = false;
164 }
165
166 // Set tolerance window for CAM_CRK synchronization in case of start
167 tolerance_window_CAM[camId] = 2.0 * revolution_CRK;
168 }
169 Stalling_detection();
170
171 configuration_complete = true;
172 }
